int main() {
	int p2, p3, p5, min, a[1501], n;
	p2 = p3 = p5 = 1;
	a[1] = 1;
	min = 247000000;
	cin >> n;

	for (int i = 2; i <= n; i++) {
		if (a[p2] * 2 > a[p3] * 3) min = a[p3] * 3;
		else min = a[p2] * 2;
		if (a[p5] * 5 < min) min = a[p5] * 5;
		if (a[p2] * 2 == min) p2++;
		if (a[p3] * 3 == min) p3++;
		if (a[p5] * 5 == min) p5++;
		a[i] = min;
	}
	cout << a[n] << endl;

	return 0;
}
